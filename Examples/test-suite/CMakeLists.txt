macro ( swig_and_compile TESTCASE LANGUAGE )
  set (SWIG_CXX_FLAG)
  set ( SWIG_CXX_EXTENSION ".c")
  if (1)
    set (SWIG_CXX_FLAG "-c++")
    set(SWIG_CXX_EXTENSION ".cxx") 
  endif ()

  set (SWIG_OUTPUT ${TESTCASE}${LANGUAGE}_wrap${SWIG_CXX_EXTENSION} )
  add_custom_command(OUTPUT ${SWIG_OUTPUT}
    COMMAND ${SWIG_EXECUTABLE} -${LANGUAGE} ${SWIG_CXX_FLAG} -I${CMAKE_SOURCE_DIR}/Lib -I${CMAKE_SOURCE_DIR}/Lib/${LANGUAGE} -I${CMAKE_BINARY_DIR}/Lib -o ${SWIG_OUTPUT} ${CMAKE_CURRENT_SOURCE_DIR}/${TESTCASE}.i
  )
  add_library(${TESTCASE}_${LANGUAGE} MODULE EXCLUDE_FROM_ALL ${SWIG_OUTPUT} )
#   set_target_properties(${TESTCASE}_${LANGUAGE})
  add_dependencies(check ${TESTCASE}_${LANGUAGE})
endmacro()


foreach( LANGUAGE ${LANGUAGES} ) 
  foreach( CPP_TEST_CASE ${CPP_TEST_CASES} )
    swig_and_compile(${CPP_TEST_CASE} ${LANGUAGE} TRUE)
  endforeach()
endforeach()
